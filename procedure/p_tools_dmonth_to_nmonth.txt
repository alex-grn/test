 CREATE  OR  REPLACE  FUNCTION  public.p_tools_dmonth_to_nmonth(month  date)
  RETURNS  integer
  LANGUAGE  plpgsql
AS  $function$
declare
	result              text;
        nmonth  		  integer;
begin  
        nmonth:=to_number(to_char(month,'mm'),'99');
	return(nmonth);
end;
$function$

CREATE OR REPLACE FUNCTION public.p_tools_dmonth_to_nmonth (
  month date
)
RETURNS integer AS
$body$
declare
	result       text;
    nmonth 		 integer;
begin 
    nmonth:=to_number(to_char(month,'mm'),'99');
	return(nmonth);
end;
$body$
LANGUAGE 'plpgsql'
VOLATILE
CALLED ON NULL INPUT
SECURITY INVOKER
COST 100;

ALTER FUNCTION public.p_tools_dmonth_to_nmonth (month date)
  OWNER TO magicbox;

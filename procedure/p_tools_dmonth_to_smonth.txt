 CREATE  OR  REPLACE  FUNCTION  public.p_tools_dmonth_to_smonth(month  date)
  RETURNS  text
  LANGUAGE  plpgsql
AS  $function$
declare
	result              text;
        nmonth  		  integer;
begin  
        nmonth:=to_number(to_char(month,'mm'),'99');
  	if  nmonth  =  1  then  result:='январь';
        elsif  nmonth  =  2  then  result:='февраль';
        elsif  nmonth  =  3  then  result:='март';
        elsif  nmonth  =  4  then  result:='апрель';
        elsif  nmonth  =  5  then  result:='март';
        elsif  nmonth  =  6  then  result:='июнь';
        elsif  nmonth  =  7  then  result:='июль';
        elsif  nmonth  =  8  then  result:='август';
        elsif  nmonth  =  9  then  result:='сентябрь';
        elsif  nmonth  =  10  then  result:='октябрь';
        elsif  nmonth  =  11  then  result:='ноябрь';
        elsif  nmonth  =  12  then  result:='декабрь';
        end  if;
	return(result);
end;
$function$

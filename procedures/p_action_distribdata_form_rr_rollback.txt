CREATE OR REPLACE FUNCTION public.p_action_distribdata_form_rr_rollback (
  idlist text
)
RETURNS void AS
$body$
declare
  rec    RECORD;
begin

  for rec in (
      select k.id/*,
             d.keyout*/
        from DISTRIBDATA k/*,
             doclinks d*/
       where k.id = any(p_system_get_selectlist(idlist))/*
         and d.tablein ilike 'DISTRIBDATA'
         and d.keyin = k.id
         and d.tableout ilike 'EXPSHD'*/
  )  
  loop
     --  perform p_system_mdl_table('delete','expshd','{"id":'||rec.keyout||'}',NULL,UID);
     delete from EXPSHD j where exists (select 1
                                             from DOCLINKS d
                                            where d.TABLEIN ilike 'DISTRIBDATA'
                                              and d.KEYIN = rec.ID
                                              and d.TABLEOUT ilike 'EXPSHD'
                                              and d.KEYOUT = j.ID);
     PERFORM P_SYSTEM_DOCLINKS_DEL('DISTRIBDATA',rec.ID,'EXPSHD');
     PERFORM P_ACTION_SET_STATUS('DISTRIBDATA',rec.ID::text,'0');
  end loop;
  
end;
$body$
LANGUAGE 'plpgsql'
VOLATILE
CALLED ON NULL INPUT
SECURITY INVOKER
COST 100;

ALTER FUNCTION public.p_action_distribdata_form_rr_rollback (idlist text)
  OWNER TO magicbox;

CREATE OR REPLACE FUNCTION public.p_action_payselect_move (
  id bigint
)
RETURNS void AS
$body$
DECLARE 
BEGIN
	update infopaydocbuff b set statusmove = '1' where b.id = p_action_payselect_move.id;
END;
$body$
LANGUAGE 'plpgsql'
VOLATILE
CALLED ON NULL INPUT
SECURITY INVOKER
COST 100;

  OWNER TO magicbox;

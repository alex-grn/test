 CREATE  OR  REPLACE  FUNCTION  public.p_tools_nmonth_to_smonth(month  integer)
  RETURNS  text
  LANGUAGE  plpgsql
AS  $function$
declare
	result              text;
begin  
  	if  month  =  1  then  result:='январь';
        elsif  month  =  2  then  result:='февраль';
        elsif  month  =  3  then  result:='март';
        elsif  month  =  4  then  result:='апрель';
        elsif  month  =  5  then  result:='март';
        elsif  month  =  6  then  result:='июнь';
        elsif  month  =  7  then  result:='июль';
        elsif  month  =  8  then  result:='август';
        elsif  month  =  9  then  result:='сентябрь';
        elsif  month  =  10  then  result:='октябрь';
        elsif  month  =  11  then  result:='ноябрь';
        elsif  month  =  12  then  result:='декабрь';
        end  if;
	return(result);
end;
$function$

 CREATE  OR  REPLACE  FUNCTION  public.p_tools_type_int(sval  text,  siso  text)
  RETURNS  text
  LANGUAGE  plpgsql
AS  $function$
declare

begin
  if  SVAL  =  '1e.'
		then
			if  SISO  =  'RUB'  --  рубль
			then
				return  '  рубль  ';
			elsif  SISO  =  'USD'  --  доллар
			then
				return  '  доллар  ';
			elsif  SISO  =  'EUR'  --  евро
			then
				return  '  евро  ';
			elsif  SISO  =  'JPY'  --  японская  иена
			then
				return  '  японский  иен  ';
			elsif  SISO  =  'GBP'  --  фунт  стерлинг
			then
				return  '  фунт  стерлинг  ';
			elsif  SISO  =  'CNY'  --  китайская  юань
			then
				return  '  китайская  юань  ';
			else
				raise  using  message  =  'Код  валюты  "%s",  не  пределен!';
			end  if;
		elsif  SVAL  =  '2e.'
		then
			if  SISO  =  'RUB'  --  рубль
			then
				return  '  рубля  ';
			elsif  SISO  =  'USD'  --  доллар
			then
				return  '  доллара  ';
			elsif  SISO  =  'EUR'  --  евро
			then
				return  '  евро  ';
			elsif  SISO  =  'JPY'  --  японская  иена
			then
				return  '  японских  иена  ';
			elsif  SISO  =  'GBP'  --  фунт  стерлинг
			then
				return  '  фунта  стерлинга  ';
			elsif  SISO  =  'CNY'  --  китайская  юань
			then
				return  '  китайских  юаня  ';
			else
				raise  using  message  =  'Код  валюты  "%s",  не  пределен!';
			end  if;
		elsif  SVAL  =  '3e.'
		then
			if  SISO  =  'RUB'  --  рубль
			then
				return  '  рубля  ';
			elsif  SISO  =  'USD'  --  доллар
			then
				return  '  доллара  ';
			elsif  SISO  =  'EUR'  --  евро
			then
				return  '  евро  ';
			elsif  SISO  =  'JPY'  --  японская  иена
			then
				return  '  японских  иена  ';
			elsif  SISO  =  'GBP'  --  фунт  стерлинг
			then
				return  '  фунта  стерлинга  ';
			elsif  SISO  =  'CNY'  --  китайская  юань
			then
				return  '  китайских  юаня  ';
			else
				raise  using  message  =  'Код  валюты  "%s",  не  пределен!';
			end  if;
		elsif  SVAL  =  '4e.'
		then
			if  SISO  =  'RUB'  --  рубль
			then
				return  '  рубля  ';
			elsif  SISO  =  'USD'  --  доллар
			then
				return  '  доллара  ';
			elsif  SISO  =  'EUR'  --  евро
			then
				return  '  евро  ';
			elsif  SISO  =  'JPY'  --  японская  иена
			then
				return  '  японских  иена  ';
			elsif  SISO  =  'GBP'  --  фунт  стерлинг
			then
				return  '  фунта  стерлинга  ';
			elsif  SISO  =  'CNY'  --  китайская  юань
			then
				return  '  китайских  юаня  ';
			else
				raise  using  message  =  'Код  валюты  "%s",  не  пределен!';
			end  if;
		elsif  SVAL  =  '.'
		then
			if  SISO  =  'RUB'  --  рубль
			then
				return  'рублей  ';
			elsif  SISO  =  'USD'  --  доллар
			then
				return  'долларов  ';
			elsif  SISO  =  'EUR'  --  евро
			then
				return  'евро  ';
			elsif  SISO  =  'JPY'  --  японская  иена
			then
				return  'японских  иен  ';
			elsif  SISO  =  'GBP'  --  фунт  стерлинг
			then
				return  'фунтов  стерлингов  ';
			elsif  SISO  =  'CNY'  --  китайская  юань
			then
				return  'китайских  юаней  ';
			else
				raise  using  message  =  'Код  валюты  "%s",  не  пределен!';
			end  if;
		end  if;
end;
$function$
